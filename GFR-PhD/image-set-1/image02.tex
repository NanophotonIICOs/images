% Wave Function

\begin{scope}[x={(-150:0.7)}, y={(90:1.0)}, z={(-15:8mm)},rotate=65,yshift=-5cm,xshift=-7cm]
	
\def\wave{
	\draw[fill, very thick, fill opacity=.2]
	     (0,0) sin (1,1) cos (2,0) sin (3,-1) cos (4,0)
	           sin (5,1) cos (6,0) sin (7,-1) cos (8,0)
	           sin (9,1) cos (10,0) sin (11,-1) cos (12,0);
         
	\foreach \shift in {0,4,8}
	{
		\begin{scope}[xshift=\shift cm,thin]
		        \draw[-stealth, thick] (.5,0)  -- (0.5,0 |- 45:1cm);
		        \draw[-stealth, thick] (1,0)   -- (1,1);
		        \draw[-stealth, thick] (1.5,0) -- (1.5,0 |- 45:1cm);
		        \draw[-stealth, thick] (2.5,0) -- (2.5,0 |- -45:1cm);
		        \draw[-stealth, thick] (3,0)   -- (3,-1);
		        \draw[-stealth, thick] (3.5,0) -- (3.5,0 |- -45:1cm);
		 \end{scope}
	} 
}

% Red Wave
\begin{scope}[canvas is zy plane at x=0, draw=red, fill=red] 
	\draw[-latex, thick, black] (0,0) -- (0, 1.5) node[above] {$\mathbf E$};
	\wave
\end{scope}

% Blue Wave
\begin{scope}[canvas is zx plane at y=0, draw=blue, fill=blue]
	%% Direction of Propagation
	\draw[-latex, thick, black] (0,0) -- (0,2) node[left] {$\mathbf B$};
	\wave
\end{scope}

\begin{scope}[x=(-15:0.9), y=(90:0.9), z=(-150:1.1),

\large
\def\A{1.5}
\def\om{1.3}
\def\nNodes{5} % use even number
\def\nVectorsPerNode{8}
\def\N{\nNodes*40}
\def\xmax{\nNodes*pi/2*1.01}
\pgfmathsetmacro\nVectors{\nVectorsPerNode*\nNodes}
\def\vE{\mathbf{E}}
\def\vB{\mathbf{B}}


% waves
\draw[Ecol,very thick,variable=\t,domain=0:\nNodes*pi/2*1.03,samples=\N]
plot (\t,{\A*cos(\om*\t*360/pi)},{\A*sin(\om*\t*360/pi)});

% draw vectors
\foreach \k [evaluate={\t=\k*pi/2/\nVectorsPerNode;
		 \angle=\k*90/\nVectorsPerNode;}]
in {1,...,\nVectors}{
\draw[EVcol,vector] (\t,0,0) -- ++(0,{\A*cos(\om*2*\angle)},{\A*sin(\om*2*\angle)});
}
